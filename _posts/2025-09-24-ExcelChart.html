import matplotlib.pyplot as plt
import numpy as np 
# Create an array of x values
x = np.linspace(-10, 10, 100)  # 100 points from -10 to 10

# Define the function
y = x**3
plt.plot(x, y, label='y = x^3', color='green')  # line plot
plt.title('Plot of y = x^3')                   # title
plt.xlabel('x')                                # x-axis label
plt.ylabel('y')                                # y-axis label
plt.legend()                                   # show legend
plt.grid(True)                                 # add grid
plt.show()                                     # display plot
# Example data
categories = ['A', 'B', 'C', 'D']
values = [10, 20, 15, 25]

plt.bar(categories, values, color='green')
plt.title('My Simple Bar Chart')
plt.xlabel('Category')
plt.ylabel('Value')
plt.show()

import pandas as pd
import matplotlib.pyplot as plt

# Load Excel
df = pd.read_excel('Cancer_Cases_deaths_2025.xlsx', sheet_name='Sheet2')

# Preview data
df.head()
# Remove rows where 'Common Type of Cancers' is null or '-'
df = df[df['Common Type of Cancers'].notna()]
df = df[df['Common Type of Cancers'] != '-']

# Optional: drop the 'Rank' column
df = df.drop(columns=['Rank'])
labels = df['Common Type of Cancers']
sizes = df['Estimated New Cases 2025']

plt.figure(figsize=(10,8))
plt.pie(sizes, labels=labels, autopct='%1.1f%%', startangle=140)
plt.title('Estimated New Cancer Cases in 2025 by Type')
plt.show()
sizes_deaths = df['Estimated Deaths 2025']

plt.figure(figsize=(10,8))
plt.pie(sizes_deaths, labels=labels, autopct='%1.1f%%', startangle=140)
plt.title('Estimated Cancer Deaths in 2025 by Type')
plt.show()

import pandas as pd
import matplotlib.pyplot as plt

# Load Excel
df = pd.read_excel('Cancer_Cases_deaths_2025.xlsx', sheet_name='Sheet2')

# Remove rows where 'Common Type of Cancers' is null or '-'
df = df[df['Common Type of Cancers'].notna()]
df = df[df['Common Type of Cancers'] != '-']

# Drop the Rank column
df = df.drop(columns=['Rank'])

# Convert numbers to integers

plt.figure(figsize=(12,6))  # Set figure size
plt.bar(df['Common Type of Cancers'], df['Estimated New Cases 2025'], color='skyblue')
plt.xticks(rotation=45, ha='right')  # Rotate x labels for readability
plt.ylabel('Estimated New Cases in 2025')
plt.title('Estimated New Cancer Cases in 2025 by Type')
plt.tight_layout()  # Adjust layout to prevent cutoff
plt.show()
plt.figure(figsize=(12,6))
plt.bar(df['Common Type of Cancers'], df['Estimated Deaths 2025'], color='salmon')
plt.xticks(rotation=45, ha='right')
plt.ylabel('Estimated Deaths in 2025')
plt.title('Estimated Cancer Deaths in 2025 by Type')
plt.tight_layout()
plt.show()
